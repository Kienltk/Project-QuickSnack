create database quick_snack_db;


-- CREATE TABLE --
	-- Independent
create table user (
	user_id int primary key auto_increment,
    fullname varchar(50) NOT NULL,
    username varchar(50) NOT NULL,
    email varchar(100) NOT NULL,
    gender varchar(1) NOT NULL,
    password_hash varchar(255) NOT NULL
);

create table category(
	category_id  int primary key auto_increment,
	category_name varchar(100)
);

create table ingredients (
	ingredient_id int auto_increment primary key,
	ingredient_name varchar(100)
);

	-- Depend
create table quick_snack(
	quick_snack_id int primary key auto_increment,
	name varchar(100) NOT NULL,
	level varchar(50) NOT NULL,
	time varchar(50) NOT NULL,
	yield varchar(100) NOT NULL,
	created_at date NOT NULL,
	user_id int NOT NULL,
	foreign key (user_id) references user(user_id)
);

create table nutrition(
	nutrition_id int auto_increment primary key,
	quick_snack_id int,
	calories varchar(15),
	fat varchar(15),
	protein varchar(15),
	carbs varchar(15),
	fiber varchar(15),
	net_crabs varchar(15),
	sodium varchar(15),
	cholesterol varchar(15),
	foreign key (quick_snack_id) references quick_snack(quick_snack_id)
);

create table recipe (
    recipe_id int auto_increment primary key,
    quick_snack_id int,
    step int,
    derection text,
    foreign key (quick_snack_id) references quick_snack(quick_snack_id)
);

create table ingredient_to_quick_snack(
	quick_snack_id int,
	ingredient_id int,
	quantity text,
	foreign key (ingredient_id) references ingredients(ingredient_id),
	foreign key (quick_snack_id) references quick_snack(quick_snack_id)
);

create table image_quick_snack (
    id int auto_increment primary key,
    quick_snack_id int,
    name_image varchar(100),
    kind boolean,
    address_img varchar(255),
    foreign key (quick_snack_id) references quick_snack(quick_snack_id)
);

create table image_category (
    id int auto_increment primary key,
    category_id int,
    name varchar(100),
    address_img_category varchar(255),
    foreign key (category_id) references category(category_id)
);

create table category_to_quick_snack(
	quick_snack_id int,
	category_id int,
	foreign key (category_id) references category(category_id),
	foreign key (quick_snack_id) references quick_snack(quick_snack_id) 
);

create table review (
	review_id int auto_increment primary key,
	quick_snack_id int,
	user_id int,
	rating int,
	comment text,
	time timestamp default current_timestamp,
	foreign key (quick_snack_id) references quick_snack(quick_snack_id),
	foreign key (user_id) references user(user_id)
);

create table user_category (
	user_category_id int primary key auto_increment,
    user_category_name varchar(50) NOT NULL,
    user_id int NOT NULL,
    foreign key (user_id) references user(user_id)
);

create table quick_snack_to_user_category(
	quick_snack_id int,
	user_category_id int,
	foreign key (quick_snack_id) references quick_snack(quick_snack_id),
	foreign key (user_category_id) references user_category(user_category_id)
);
