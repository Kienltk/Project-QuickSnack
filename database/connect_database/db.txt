create database quick_snack_db;

use quick_snack_db;
-- CREATE TABLE --
	-- Independent
create table user (
	user_id int primary key auto_increment,
    fullname varchar(50) NOT NULL,
    username varchar(50) NOT NULL,
    email varchar(100) NOT NULL,
    gender varchar(1) NOT NULL,
    password_hash varchar(255) NOT NULL
);

create table category(
	category_id int primary key auto_increment,
	category_name varchar(100)
);

create table ingredients (
	ingredient_id int auto_increment primary key,
	ingredient_name varchar(100)
);

create table banner (
	banner_id int auto_increment primary key,
	banner_name varchar(255),
    address_banner_img varchar(255) 
);



	-- Depend
create table quick_snack(
	quick_snack_id int primary key auto_increment,
	name varchar(100) NOT NULL,
	level varchar(50) NOT NULL,
	time varchar(50) NOT NULL,
	yield varchar(100) NOT NULL,
	created_at date NOT NULL,
	user_id int NOT NULL,
	foreign key (user_id) references user(user_id)
);

create table nutrition(
	nutrition_id int auto_increment primary key,
	quick_snack_id int,
	calories varchar(15),
	fat varchar(15),
	protein varchar(15),
	carbs varchar(15),
	fiber varchar(15),
	net_crabs varchar(15),
	sodium varchar(15),
	cholesterol varchar(15),
	foreign key (quick_snack_id) references quick_snack(quick_snack_id)
);

create table recipe (
    recipe_id int auto_increment primary key,
    quick_snack_id int,
    step int,
    derection text,
    foreign key (quick_snack_id) references quick_snack(quick_snack_id)
);

create table ingredient_to_quick_snack(
	quick_snack_id int,
	ingredient_id int,
	quantity text,
	foreign key (ingredient_id) references ingredients(ingredient_id),
	foreign key (quick_snack_id) references quick_snack(quick_snack_id)
);

create table image_quick_snack (
    id int auto_increment primary key,
    quick_snack_id int,
    name_image varchar(100),
    kind boolean,
    address_img varchar(255),
    foreign key (quick_snack_id) references quick_snack(quick_snack_id)
);

create table image_category (
    id int auto_increment primary key,
    category_id int,
    name varchar(100),
    address_img_category varchar(255),
    foreign key (category_id) references category(category_id)
);

create table category_to_quick_snack(
	quick_snack_id int,
	category_id int,
	foreign key (category_id) references category(category_id),
	foreign key (quick_snack_id) references quick_snack(quick_snack_id) 
);

create table review (
	review_id int auto_increment primary key,
	quick_snack_id int,
	user_id int,
	rating int,
	comment text,
	time timestamp default current_timestamp,
	foreign key (quick_snack_id) references quick_snack(quick_snack_id),
	foreign key (user_id) references user(user_id)
);

create table user_category (
	user_category_id int primary key auto_increment,
    user_category_name varchar(50) NOT NULL,
    user_id int NOT NULL,
    foreign key (user_id) references user(user_id)
);

create table quick_snack_to_user_category(
	quick_snack_id int,
	user_category_id int,
	foreign key (quick_snack_id) references quick_snack(quick_snack_id),
	foreign key (user_category_id) references user_category(user_category_id)
);


CREATE TABLE contact (
	id INT AUTO_INCREMENT PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    email VARCHAR(100) NOT NULL,
    phone VARCHAR(20) NOT NULL,
    message TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
 );

-- QUERY SELECT TABLE --
SELECT quick_snack_id, address_img FROM image_quick_snack WHERE kind = 0 ORDER BY RAND() LIMIT 1;
select * from ingredients;
select * from category;
SELECT * FROM image_quick_snack WHERE quick_snack_id = 3 AND kind = 1 LIMIT 1;
SELECT quick_snack_id, address_img FROM image_quick_snack WHERE kind = 1 ORDER BY RAND() LIMIT 10;

SELECT 
    qs.quick_snack_id,
    qs.name,
    qs.level,
    qs.time,
    qs.yield,
    qs.created_at,
    qs.user_id,
    MAX(iqs.address_img) AS image_address,
    GROUP_CONCAT(DISTINCT c.category_name ORDER BY c.category_id SEPARATOR ', ') AS categories,
    ROUND(AVG(r.rating), 1) AS average_rating
FROM 
    quick_snack qs
LEFT JOIN 
    image_quick_snack iqs ON qs.quick_snack_id = iqs.quick_snack_id
LEFT JOIN 
    category_to_quick_snack ctqs ON qs.quick_snack_id = ctqs.quick_snack_id
LEFT JOIN 
    category c ON ctqs.category_id = c.category_id AND c.category_id IS NOT NULL
LEFT JOIN 
    review r ON qs.quick_snack_id = r.quick_snack_id
GROUP BY 
    qs.quick_snack_id,
    qs.name,
    qs.level,
    qs.time,
    qs.yield,
    qs.created_at,
    qs.user_id
ORDER BY 
    qs.quick_snack_id;
    
SET sql_mode='STRICT_TRANS_TABLES,NO_ZERO_IN_DATE,NO_ZERO_DATE,ERROR_FOR_DIVISION_BY_ZERO,NO_ENGINE_SUBSTITUTION';
